if bmi <= 15:
    print('Very severely underweight')
elif bmi <= 16:
    print('Severely underweight')
elif bmi <= 18.5:
    print('Underweight')
elif bmi <= 25:
    print('Normal (healthy weight)')

    print('Overweight')

    print('Obese Class I (Moderately obese)')
