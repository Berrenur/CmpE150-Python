  ```python
def is_perfect(num):
    sum_of_divisors = 0
    for i in range(1, num):
        if num % i == 0:
            sum_of_divisors += i
    return sum_of_divisors == num

def all_the_perfects(N, M):
    for i in range(N, M + 1):
        if is_perfect(i):
            print(i, end=" ")

all_the_perfects(1, 10000)
  ```
